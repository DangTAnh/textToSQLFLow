from text_to_sql_flow.output.html_renderer import render_html_report, HTML_TEMPLATE
from text_to_sql_flow.types import Flow, Step, StepOutput, Diagram, CreatedDate
from datetime import datetime, timezone
import tempfile, os

flow = Flow(
    name='test_flow',
    description='Test flow description',
    steps=[
        Step(
            name='STEP_ONE', parents=[], order=1,
            sql='SELECT * FROM source',
            output=StepOutput(tempView='source_vw'),
            diagram=Diagram(x=0, y=0),
            createdDate=CreatedDate(date=datetime.now(timezone.utc))
        ),
    ]
)
with tempfile.TemporaryDirectory() as tmp:
    path = render_html_report(flow, tmp)
    assert path.exists()
    content = open(path).read()
    assert 'test_flow' in content
    assert 'STEP_ONE' in content
    assert '</html>' in content
    print(f'HTML report OK: {path}')
