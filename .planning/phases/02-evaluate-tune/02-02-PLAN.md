    ```python
    def run_evaluation_loop(
        description: str,
        output_dir: Path,
        auto: bool = False,
        interactive: bool = False,
    ) -> Path:
        """
        Run generate-evaluate-tune loop.

        Generates a flow, evaluates it against the rubric, and if score < threshold,
        appends evaluator feedback to the description and regenerates.
        Loops until score >= THRESHOLD or MAX_ITERATIONS reached.

        Args:
            description: Business description of the Spark SQL ETL flow.
            output_dir: Directory to write output files.
            auto: If True, run without user prompts (for CI/batch).
            interactive: If True, pause at each iteration for user review.

        Returns:
            Path to the final generated JSON file.
        """
        current_description = description
        last_result: EvaluationResult | None = None

        for iteration in range(1, MAX_ITERATIONS + 1):
            logger.info("Evaluation loop iteration %d/%d", iteration, MAX_ITERATIONS)

            # Step 1: Generate
            flow_path = run_generation(current_description, output_dir)

            # Step 2: Evaluate
            try:
                result = evaluate_flow(flow_path)
            except Exception as e:
                logger.error("Evaluation failed: %s", e)
                # If evaluation fails on the last iteration, return whatever we have
                if iteration == MAX_ITERATIONS:
                    return flow_path
                continue

            last_result = result

            # Step 3: Check if passed
            if result.passed:
                logger.info(
                    "Flow passed evaluation: score=%.1f/10 (threshold=%.1f)",
                    result.score,
                    THRESHOLD,
                )
                return flow_path

            logger.info(
                "Score %.1f/10 below threshold %.1f (iteration %d/%d)",
                result.score,
                THRESHOLD,
                iteration,
                MAX_ITERATIONS,
            )

            # Step 4: Interactive mode - ask user
            if interactive:
                _show_interactive_prompt(result, iteration)
                action = _get_interactive_action()
                if action == "abort":
                    logger.info("User aborted evaluation loop")
                    return flow_path
                elif action == "continue":
                    logger.info("User chose to continue with current output")
                    return flow_path
                # action == "retry": fall through to tune

            # Step 5: Tune prompt with feedback (per EVAL-03)
            if iteration < MAX_ITERATIONS:
                current_description = _tune_prompt(current_description, result.feedback)
                logger.info("Tuned prompt with evaluator feedback (iteration %d -> %d)", iteration, iteration + 1)

        # Max iterations reached, return last result
        logger.warning(
            "Max iterations (%d) reached. Last score: %.1f/10",
            MAX_ITERATIONS,
            last_result.score if last_result else 0,
        )
        return flow_path if last_result is None else _regenerate_path(output_dir)
    ```

    ```python
    def _show_interactive_prompt(result: EvaluationResult, iteration: int) -> None:
        typer.echo("")
        typer.echo("=" * 50)
        typer.echo(f"Iteration {iteration}/{MAX_ITERATIONS} — Evaluation Result")
        typer.echo("=" * 50)
        typer.echo(f"Overall Score: {result.score:.1f}/10 (threshold: {THRESHOLD}/10)")
        typer.echo("")
        typer.echo("Dimensions:")
        for dim, score in sorted(result.dimensions.items()):
            bar = "█" * int(score) + "░" * (10 - int(score))
            typer.echo(f"  {dim.replace('_', ' ').title():25s} {bar} {score:.0f}/10")
        typer.echo("")
        typer.echo("Feedback:")
        typer.echo(f"  {result.feedback}")
        typer.echo("")
    ```

    `_get_interactive_action() -> str`:
    ```python
    def _get_interactive_action() -> str:
        typer.echo("Options:")
        typer.echo("  [1] Retry with feedback (auto-tune)")
        typer.echo("  [2] Abort (discard current output)")
        typer.echo("  [3] Continue (use as-is)")
        typer.echo("")
        while True:
            choice = typer.prompt("Choose (1-3)", default="1")
            if choice in ("1", "2", "3"):
                return {"1": "retry", "2": "abort", "3": "continue"}[choice]
            typer.echo("Invalid choice. Enter 1, 2, or 3.")
    ```

    ```python
    @app.command()
    def generate(
        description: str = typer.Argument(
            ...,
            help="Business description of the Spark SQL ETL flow",
        ),
        output: Path = typer.Option(
            "./output",
            "--output",
            "-o",
            help="Output directory for generated flow files",
            file_okay=False,
            dir_okay=True,
        ),
        auto: bool = typer.Option(
            False,
            "--auto",
            help="Run evaluation loop automatically without prompts (for CI/batch)",
        ),
        interactive: bool = typer.Option(
            False,
            "--interactive",
            help="Pause at each evaluation iteration for user review",
        ),
    ):
        """
        Generate a Spark SQL flow from a business description.

        By default, generates the flow once and writes JSON output.
        Use --auto to enable the evaluate-tune loop (auto-retry until quality threshold met).
        Use --interactive to review each evaluation result and choose retry/abort/continue.
        """
        from text_to_sql_flow.pipeline import run_generation, run_evaluation_loop

        console = Console()

        if auto or interactive:
            # Use evaluation loop with Rich progress
            with Progress(
                SpinnerColumn(),
                TextColumn("[progress.description]{task.description}"),
                BarColumn(),
                console=console,
                transient=False,
            ) as progress:
                task = progress.add_task(
                    description="[cyan]Generating flow...",
                    total=MAX_ITERATIONS,
                )
                result_path = run_evaluation_loop(
                    description=description,
                    output_dir=output,
                    auto=auto,
                    interactive=interactive,
                )
                progress.update(task, completed=MAX_ITERATIONS)
                progress.update(task, description="[green]Done!")
        else:
            # Original behavior: single generation without evaluation loop
            result_path = run_generation(description=description, output_dir=output)

        console.print(f"[green]Flow generated successfully:[/green] {result_path}")
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User description → pipeline loop | Untrusted business description enters prompt tuning |
| Interactive user input → loop | User enters continue/retry/abort choice |
| LLM evaluator response → loop | Evaluator feedback is appended to generation prompt |
| CLI flags → loop logic | --auto/--interactive flags control loop behavior |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-03 | Denial of Service | pipeline.py run_evaluation_loop | mitigate | Hard max of MAX_ITERATIONS (5) prevents infinite loop; caller (cli.py) can always Ctrl+C |
| T-02-04 | Elevation of Privilege | pipeline.py _get_interactive_action | mitigate | Input validated: only 1, 2, or 3 accepted; infinite loop on bad input is prevented by prompting again |
| T-02-05 | Tampering | pipeline.py _tune_prompt | accept | Feedback from LLM evaluator is appended to user message role, not system; no prompt injection risk elevation beyond existing generation prompt |
| T-02-SC | Tampering | rich package | mitigate | rich >=13.0 is a well-known [ASSUMED] Python package (PyPI top 100, 20k+ GitHub stars) |
</threat_model>
